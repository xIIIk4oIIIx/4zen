/// Publiczny obserwator tras nawigacyjnych.
class ZenRouteObserver extends NavigatorObserver {
  final Function(Route?) onChanged;
  ZenRouteObserver(this.onChanged);

  @override
  void didPush(Route route, Route? previousRoute) => onChanged(route);

  @override
  void didPop(Route route, Route? previousRoute) => onChanged(previousRoute);
}

/// Centralny router aplikacji zarządzający logiką przejść.
class ZenRouter {
  /// Generuje trasę na podstawie ustawień nawigacji.
  static Route<dynamic>? generateRoute(RouteSettings settings) {
    // Trasa dla odtwarzacza
    if (settings.name == PlayerScreen.routeName) {
      final track = settings.arguments as AudioTrack;
      return PageRouteBuilder(
        settings: settings,
        opaque: true,
        barrierDismissible: false,
        barrierColor: const Color(0xFF0F172A),
        transitionDuration: const Duration(milliseconds: 500),
        reverseTransitionDuration: const Duration(milliseconds: 500),
        pageBuilder: (_, __, ___) => Scaffold(
          backgroundColor: const Color(0xFF0F172A),
          body: PlayerScreen(track: track),
        ),
        transitionsBuilder: (_, animation, secondaryAnimation, child) {
          // Animacja SlideTransition: wejście od dołu do góry
          final slideAnimation = animation.drive(
            Tween<Offset>(
              begin: const Offset(0, 1),
              end: Offset.zero,
            ).chain(CurveTween(curve: Curves.easeOutCubic)),
          );

          return SlideTransition(
            position: slideAnimation,
            child: child,
          );
        },
      );
    }

    // Trasa dla konfiguracji folderu
    if (settings.name == FolderSetupPage.routeName) {
      final onFolderSelected = settings.arguments as Function(String);
      return PageRouteBuilder(
        settings: settings,
        transitionDuration: const Duration(milliseconds: 400),
        pageBuilder: (_, __, ___) => FolderSetupPage(onFolderSelected: onFolderSelected),
        transitionsBuilder: (_, animation, secondaryAnimation, child) {
          return SlideTransition(
            position: animation.drive(
              Tween<Offset>(
                begin: const Offset(0, 1),
                end: Offset.zero,
              ).chain(CurveTween(curve: Curves.easeOutCubic)),
            ),
            child: child,
          );
        },
      );
    }

    return null;
  }
}

  // Prywatna instancja dla wzorca Singleton
  static final SettingsService _instance = SettingsService._internal();

  /// Zwraca jedyną instancję klasy SettingsService.
  factory SettingsService() => _instance;

  SettingsService._internal();

  late SharedPreferences _prefs;

  /// Inicjalizuje instancję SharedPreferences.
  /// Metoda powinna zostać wywołana przy starcie aplikacji.
  Future<void> init() async {
    try {
      _prefs = await SharedPreferences.getInstance();
      debugPrint("SettingsService: Inicjalizacja zakończona pomyślnie.");
    } catch (e) {
      debugPrint("SettingsService: Błąd podczas inicjalizacji SharedPreferences: $e");
      rethrow;
    }
  }

  /// Getter zwracający zapisaną ścieżkę folderu medytacji.
  /// Zwraca null, jeśli ścieżka nie została wcześniej zapisana.
  String? get folderPath {
    try {
      return _prefs.getString(_folderPathKey);
    } catch (e) {
      debugPrint("SettingsService: Błąd podczas pobierania folderPath: $e");
      return null;
    }
  }

  /// Zapisuje ścieżkę folderu medytacji w pamięci trwałej urządzenia.

  /// [path] - Pełna ścieżka do katalogu wybranego przez użytkownika.

    try {
      await _prefs.setString(_folderPathKey, path);
      debugPrint("SettingsService: Pomyślnie zapisano ścieżkę folderu: $path");
    } catch (e) {
      debugPrint("SettingsService: Błąd podczas zapisywania folderPath: $e");
    }
  }
}

class FileRepository {
  final DatabaseService _dbService = DatabaseService();
  final AudioPlayer _scanPlayer = AudioPlayer();

  /// Centralna logika zarządzania uprawnień (zgodna z permission_handler 12.x).
  /// Obsługuje Android 13+ (granulowane uprawnienia audio) oraz starsze wersje (storage).
  Future<bool> checkPermissions() async {
    if (!Platform.isAndroid) return true;

    debugPrint("--- Rozpoczęto weryfikację uprawnień (Android 14 Compatibility) ---");
    // 1. Wymuszone sprawdzenie powiadomień (krytyczne dla serwisu audio)
    final notificationStatus = await Permission.notification.request();
    debugPrint("Permission.notification: $notificationStatus");

    // 2. Logika dla Android 13+ (SDK 33) - Permission.audio
    PermissionStatus audioStatus = await Permission.audio.status;
    if (!audioStatus.isGranted) {
      audioStatus = await Permission.audio.request();
      debugPrint("Request Permission.audio: $audioStatus");
    }

    if (audioStatus.isGranted) {
      debugPrint("Przyznano uprawnienia Audio (Android 13+).");
      return true;
    }

    // 3. Fallback dla starszych wersji Androida (< SDK 33) - Permission.storage
    debugPrint("Próba uzyskania dostępu poprzez Permission.storage (Legacy Fallback)...");
    PermissionStatus storageStatus = await Permission.storage.status;

    if (!storageStatus.isGranted) {
      storageStatus = await Permission.storage.request();
      debugPrint("Request Permission.storage: $storageStatus");
    }

    final isGranted = storageStatus.isGranted || audioStatus.isGranted;
    debugPrint("Ostateczny wynik uprawnień: ${isGranted ? 'SUKCES' : 'BRAK DOSTĘPU'}");
    return isGranted;
  }

  /// Pobiera listę playlist na podstawie cache'u lub skanowania plików.
  /// Implementuje mechanizm Cache-First i asynchroniczną synchronizację.
  Future<List<Playlist>> getPlaylists({String? customPath}) async {
    debugPrint("--- Rozpoczęto proces pobierania playlist ---");
    List<Playlist> cachedPlaylists;
    try {
      // 1. Pobranie danych z cache
      cachedPlaylists = await _dbService.getCachedPlaylists();
    } catch (e) {
      debugPrint("FileRepository Error: Błąd podczas pobierania danych z bazy danych: $e");
      // Propagujemy błąd dalej, zamiast zwracać cichą pustą listę
      rethrow;
    }

    if (cachedPlaylists.isNotEmpty && customPath == null) {
      debugPrint("Zwracanie danych z cache (${cachedPlaylists.length} playlist). Synchronizacja w tle...");
      // 2. Synchronizacja w tle bez blokowania UI i bez sztucznego opóźnienia
      _syncFileSystemWithCache();
      return cachedPlaylists;
    }

    // 3. Jeśli cache pusty lub wymuszona ścieżka, wykonaj pełne skanowanie
    debugPrint("Cache pusty lub wymuszona ścieżka. Rozpoczynanie pełnego skanowania...");
    return await _syncFileSystemWithCache(customPath: customPath);
  }

  /// Skanuje system plików i aktualizuje bazę danych.
  Future<List<Playlist>> _syncFileSystemWithCache({String? customPath}) async {
    debugPrint("FileRepository: Rozpoczęto _syncFileSystemWithCache");

    if (!(await checkPermissions())) return [];

    final savedPath = SettingsService().folderPath;
    final targetPath = customPath ?? savedPath;
    if (targetPath == null) {
      debugPrint("Błąd synchronizacji: Brak zdefiniowanej ścieżki folderu.");
      return [];
    }

    try {
      Directory rootDir = Directory(targetPath);
      if (!await rootDir.exists()) {
        debugPrint("Błąd synchronizacji: Katalog docelowy nie istnieje: ${rootDir.path}");
        return [];
      }

      final currentCache = await _dbService.getCachedPlaylists();
      final Map<String, Map<String, dynamic>> cachedTracksData = {
        for (var p in currentCache)
          for (var t in p.tracks)
            t.id: {
              'id': t.id,
              'title': t.title,
              'filePath': t.filePath,
              'duration_ms': t.duration?.inMilliseconds,
            }
      };

      // Wywołanie izolatu do ciężkiej operacji skanowania i odczytu tagów
      // Zwraca bezpośrednio listę obiektów Playlist
      final List<Playlist> playlists = await compute(_scanTask, {
        'directoryPath': targetPath,
        'cachedTracksData': cachedTracksData,
      });

      try {
        await _dbService.clearCache();
        await _dbService.savePlaylists(playlists);
        debugPrint("Synchronizacja zakończona. Cache zaktualizowany.");
      } catch (dbError) {
        debugPrint("Błąd krytyczny podczas aktualizacji bazy danych w _syncFileSystemWithCache: $dbError");
        // Kontynuujemy, aby zwrócić przynajmniej to, co zostało przeskanowane do UI
      }

      debugPrint("FileRepository: Zakończono _syncFileSystemWithCache");
      return playlists;
    } catch (e) {
      debugPrint("Błąd krytyczny w _syncFileSystemWithCache: $e");
      return [];
    }
  }

  /// Zadanie wykonywane w osobnym izolacie
  static Future<List<Playlist>> _scanTask(Map<String, dynamic> params) async {
    final String directoryPath = params['directoryPath'];
    final Map<String, dynamic> cachedTracksData = params['cachedTracksData'];

    final rootDir = Directory(directoryPath);
    List<Playlist> playlists = [];
    List<File> looseFiles = [];

    List<FileSystemEntity> entities = [];
    try {
      entities = rootDir.listSync(recursive: false);
    } catch (e) {
      debugPrint("FileRepository Isolate Error: Błąd dostępu do folderu głównego: $e");
      return [];
    }

    for (var entity in entities) {
      if (entity is Directory) {
        final tracks = await _scanDirectory(entity, cachedTracksData);
        if (tracks.isNotEmpty) {
          playlists.add(Playlist(
            id: entity.path,
            title: p.basename(entity.path),
            tracks: tracks,
          ));
        }
      } else if (entity is File && entity.path.toLowerCase().endsWith('.mp3')) {
        looseFiles.add(entity);
      }
    }

    if (looseFiles.isNotEmpty) {
      List<AudioTrack> looseTracks = [];
      for (var file in looseFiles) {
        looseTracks.add(await _processFile(file, cachedTracksData));
      }
      playlists.add(Playlist(
        id: 'loose_tracks_root',
        title: 'Inne / Pobrane',
        tracks: looseTracks,
      ));
    }

    return playlists;
  }

  static Future<List<AudioTrack>> _scanDirectory(Directory dir, Map<String, dynamic> cachedTracksData) async {
    List<AudioTrack> tracks = [];

    List<FileSystemEntity> entities = [];
    try {
      entities = dir.listSync();
    } catch (e) {
      debugPrint("FileRepository Isolate Error: Błąd dostępu do folderu ${dir.path}: $e");
      return [];
    }

    final files = entities.whereType<File>().where((f) => f.path.toLowerCase().endsWith('.mp3'));

    for (var file in files) {
      try {
        tracks.add(await _processFile(file, cachedTracksData));
      } catch (e) {
        debugPrint("FileRepository: Błąd podczas skanowania pliku ${file.path}: $e");
      }
    }
    return tracks;
  }

  static Future<AudioTrack> _processFile(File file, Map<String, dynamic> cachedTracksData) async {
    String filename = p.basename(file.path);
    String title = filename.replaceAll(RegExp(r'\.mp3$', caseSensitive: false), '');

    // Walidacja rozmiaru pliku (Safe-Fail dla uszkodzonych zasobów)
    if (await file.length() == 0) {
      return AudioTrack(
        id: file.path,
        title: title,
        filePath: file.path,
        duration: null,
      );
    }

    if (cachedTracksData.containsKey(file.path)) {
      final cached = cachedTracksData[file.path];
      return AudioTrack(
        id: cached['id'],
        title: cached['title'],
        filePath: cached['filePath'],
        duration: cached['duration_ms'] != null ? Duration(milliseconds: cached['duration_ms']) : null,
      );
    }

    int? durationMs;
    try {
      final tags = await AudioTags.read(file.path);
      if (tags != null) {
        if (tags.title != null && tags.title!.isNotEmpty) {
          title = tags.title!;
        }
        if (tags.duration != null) {
          durationMs = (tags.duration! * 1000).toInt();
        }
      }
    } catch (e) {
      debugPrint("FileRepository: Błąd odczytu tagów dla ${file.path}: $e");
      return AudioTrack(
        id: file.path,
        title: title,
        filePath: file.path,
        duration: null,
      );
    }

    return AudioTrack(
      id: file.path,
      title: title,
      filePath: file.path,
      duration: durationMs != null ? Duration(milliseconds: durationMs) : null,
    );
  }

  void dispose() {
    _scanPlayer.dispose();
  }
}

  void _initListeners() {
    if (_player == null) return;

    for (var sub in _appSubscriptions) sub.cancel();
    _appSubscriptions.clear();

    // Mapowanie zdarzeń odtwarzacza na stan AudioHandler
    _appSubscriptions.add(_player!.playbackEventStream
        .map(_transformEvent)
        .listen(playbackState.add));

    // Mapowanie sekwencji na MediaItem
    _appSubscriptions.add(_player!.sequenceStateStream.listen((state) {
      final sequence = state?.sequence;
      if (sequence == null || sequence.isEmpty) return;
      final index = state?.currentIndex;
      if (index == null) return;
      final source = sequence[index];
      if (source.tag is MediaItem) {
        mediaItem.add(source.tag as MediaItem);
      }
    }));

    _appSubscriptions.add(_player!.currentIndexStream.listen((index) {
      if (_player == null) return;
      if (index == null) return;

      if (_stopAtEndOfTrack == true && _lastIndex != null && index != _lastIndex) {
        pause();
        _stopAtEndOfTrack = false;
        setSleepTimer(null);
        debugPrint("ZenAudioService: Zatrzymano na końcu utworu.");
      }

      if (index < _playlistTracks.length) {
        _currentTrack = _playlistTracks[index];
        _currentTrackController.add(_currentTrack);
      }

      _lastIndex = index;
    }));

    _appSubscriptions.add(_player!.positionStream.listen((position) {
      if (_player == null) return;
      final duration = _player!.duration;
      final track = _currentTrack;
      if (duration != null && track != null && duration > Duration.zero) {
        if (position >= duration - const Duration(seconds: 1)) {
          markAsCompleted(track.id);
        }
      }
    }));

    _appSubscriptions.add(_player!.playerStateStream.listen((state) {
      if (_player == null) return;
      if (state.processingState == ProcessingState.completed) {
        if (_currentTrack != null) {
          markAsCompleted(_currentTrack!.id);
        }
      }
    }));
  }

  PlaybackState _transformEvent(PlaybackEvent event) {
    return PlaybackState(
      controls: [
        MediaControl.skipToPrevious,
        if (_player!.playing) MediaControl.pause else MediaControl.play,
        MediaControl.stop,
        MediaControl.skipToNext,
      ],
      systemActions: const {
        MediaAction.seek,
        MediaAction.seekForward,
        MediaAction.seekBackward,
      },
      androidCompactActionIndices: const [0, 1, 3],
      processingState: const {
        ProcessingState.idle: AudioProcessingState.idle,
        ProcessingState.loading: AudioProcessingState.loading,
        ProcessingState.buffering: AudioProcessingState.buffering,
        ProcessingState.ready: AudioProcessingState.ready,
        ProcessingState.completed: AudioProcessingState.completed,
      }[event.processingState]!,
      playing: _player!.playing,
      updatePosition: _player!.position,
      bufferedPosition: _player!.bufferedPosition,
      speed: _player!.speed,
      queueIndex: event.currentIndex,
    );
  }

  @override
  Future<void> stop() async {
    await HapticFeedback.mediumImpact();

    // KROK 1: Anuluj subskrypcje
    for (var sub in _appSubscriptions) sub.cancel();
    _appSubscriptions.clear();

    // KROK 2: Ustaw _isInitialized = false
    _isInitialized = false;
    _initCompleter = null;

    // KROK 3: Zatrzymaj odtwarzacze
    await _player?.stop();
    await _ambientPlayer?.stop();

    // Czyszczenie timerów
    _sleepTimer?.cancel();
    _countdownTimer?.cancel();
    _sleepTimer = null;
    _countdownTimer = null;

    // Aktualizacja stanów kontrolerów
    _sleepTimerRemainingController.add(null);
    _activeAmbientController.add(null);
    _currentTrackController.add(null);
    _activeAmbientId = null;
    _currentTrack = null;

    // KROK 4: Wyczyść metadane powiadomienia
    mediaItem.add(null);

    // KROK 5: Wyślij stan Idle
    playbackState.add(PlaybackState(
      playing: false,
      processingState: AudioProcessingState.idle,
      controls: [],
    ));

    // KROK 6: Wywołanie bazowej metody stop()
    await super.stop();
  }

  Future<void> _ensureInitialized() async {
    if (_isInitialized && _player != null && _ambientPlayer != null) return;

    if (_initCompleter != null) {
      return _initCompleter!.future;
    }

    _initCompleter = Completer<void>();

    try {
      final session = await AudioSession.instance;
      await session.configure(const AudioSessionConfiguration.music());
      await session.setActive(true);

      _player ??= AudioPlayer();
      if (_ambientPlayer == null) {
        _ambientPlayer = AudioPlayer();
        await _ambientPlayer!.setLoopMode(LoopMode.one);
        await _ambientPlayer!.setVolume(currentAmbientVolume);
      }

      _initListeners();
      _isInitialized = true;
      _initCompleter!.complete();
    } catch (e) {
      debugPrint("Błąd inicjalizacji sesji: $e");
      _isInitialized = false;
      _initCompleter!.completeError(e);
      _initCompleter = null;
      rethrow;
    }
  }

  Future<void> loadTrack(AudioTrack track, {String? playlistName}) async {

  Future<void> loadPlaylist(List<AudioTrack> tracks, int initialIndex, {String? playlistName}) async {
    if (tracks.isEmpty) return;
    try {
      await _ensureInitialized();
      _lastIndex = null;
      _playlistTracks = tracks;
      _currentPlaylistName = playlistName;
      final startIndex = initialIndex.clamp(0, tracks.length - 1);
      final List<AudioSource> audioSources = [];
      for (var track in tracks) {
        if (!File(track.filePath).existsSync()) continue;
        final source = AudioSource.uri(
          Uri.file(track.filePath),
          tag: MediaItem(
            id: track.filePath,
            album: playlistName ?? "Dla Zen",
            title: track.title,
            duration: track.duration,
            artUri: _localAppIconPath != null ? Uri.file(_localAppIconPath!) : null,
          ),
        );
        audioSources.add(source);
      }

      if (audioSources.isEmpty) throw Exception("Pusta kolejka.");

      final playlist = ConcatenatingAudioSource(children: audioSources);
      await _player!.setAudioSource(playlist, initialIndex: startIndex);
    } catch (e) {
      debugPrint("Błąd ładowania playlisty: $e");
      rethrow;
    }
  }

      await _player!.setLoopMode(LoopMode.one);
    } else if (currentMode == LoopMode.one) {

    } else {
      await _player!.setLoopMode(LoopMode.off);
    }
  }

  void setSleepTimer(Duration? duration) {
    HapticFeedback.lightImpact();
    _sleepTimer?.cancel();
    _countdownTimer?.cancel();
    _stopAtEndOfTrack = false;
    if (duration == null || duration == Duration.zero) {
      _sleepTimerRemainingController.add(null);
      _stopAtEndOfTrackController.add(false);
      return;
    }

    if (duration.isNegative) {
      _stopAtEndOfTrack = true;
      _sleepTimerRemainingController.add(null);
      _stopAtEndOfTrackController.add(true);
      return;
    }

    _stopAtEndOfTrackController.add(false);
    _sleepTimer = Timer(duration, () {
      pause();
      _countdownTimer?.cancel();
      _sleepTimerRemainingController.add(null);
    });
    DateTime endTime = DateTime.now().add(duration);
    _sleepTimerRemainingController.add(duration);
    _countdownTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      final remaining = endTime.difference(DateTime.now());
      if (remaining.isNegative) {
        timer.cancel();
        _sleepTimerRemainingController.add(null);
      } else {
        _sleepTimerRemainingController.add(remaining);
      }
    });
  }

import 'dart:async';
import 'package:flutter/foundation.dart';
import 'package:path/path.dart';
import 'package:sqflite/sqflite.dart';
import 'package:for_zen/features/player/domain/models/playlist.dart';
import 'package:for_zen/features/player/domain/models/audio_track.dart';

class DatabaseService {
  static final DatabaseService _instance = DatabaseService._internal();
  static Database? _database;
  static Completer<Database>? _initCompleter;

  factory DatabaseService() => _instance;

  DatabaseService._internal();

  /// Pobiera instancję bazy danych.
  /// Jeśli baza nie jest zainicjalizowana,
  /// rozpoczyna proces asynchronicznie, zapobiegając wielokrotnej inicjalizacji.
  Future<Database> get database async {
    if (_database != null) return _database!;

    if (_initCompleter != null) {
      return _initCompleter!.future;
    }

    _initCompleter = Completer<Database>();
    final stopwatch = Stopwatch()..start();

    try {
      debugPrint("DatabaseService: Rozpoczynanie asynchronicznej inicjalizacji bazy danych...");
      _database = await _initDatabase();
      stopwatch.stop();
      debugPrint("DatabaseService: Baza danych otwarta w ${stopwatch.elapsedMilliseconds}ms.");
      _initCompleter!.complete(_database);
      return _database!;
    } catch (e) {
      stopwatch.stop();
      debugPrint("DatabaseService Error: Krytyczny błąd podczas otwierania bazy danych (${stopwatch.elapsedMilliseconds}ms): $e");
      // Poprawne czyszczenie stanu w przypadku błędu, aby umożliwić ponowną próbę
      _database = null;
      _initCompleter!.completeError(e);
      _initCompleter = null;
      rethrow;
    }
  }

  Future<Database> _initDatabase() async {
    final dbPath = await getDatabasesPath();
    final path = join(dbPath, 'zen_cache.db');

    return await openDatabase(
      path,
      version: 2,
      onCreate: (db, version) async {
        await db.execute('''
          CREATE TABLE playlists (
            id TEXT PRIMARY KEY,
            title TEXT,
            path TEXT
          )
        ''');
        await db.execute('''
          CREATE TABLE tracks (
            id TEXT PRIMARY KEY,
            playlist_id TEXT,
            title TEXT,
            file_path TEXT,
            duration_ms INTEGER,
            FOREIGN KEY (playlist_id) REFERENCES playlists (id) ON DELETE CASCADE
          )
        ''');
        // Indeks przeniesiony bezpośrednio do onCreate
        await db.execute('CREATE INDEX idx_tracks_playlist_id ON tracks (playlist_id)');
      },
      onUpgrade: (db, oldVersion, newVersion) async {
        if (oldVersion < 2) {
          // Logika aktualizacji jeśli indeks nie istniał w starszych wersjach
          await db.execute('CREATE INDEX IF NOT EXISTS idx_tracks_playlist_id ON tracks (playlist_id)');
        }
      },
    );
  }

  Future<void> savePlaylists(List<Playlist> playlists) async {
    final db = await database;
    await db.transaction((txn) async {
      final batch = txn.batch();

      for (var playlist in playlists) {
        batch.insert(
          'playlists',
          {
            'id': playlist.id,
            'title': playlist.title,
            'path': playlist.id,
          },
          conflictAlgorithm: ConflictAlgorithm.replace,
        );

        for (var track in playlist.tracks) {
          batch.insert(
            'tracks',
            {
              'id': track.id,
              'playlist_id': playlist.id,
              'title': track.title,
              'file_path': track.filePath,
              'duration_ms': track.duration?.inMilliseconds,
            },
            conflictAlgorithm: ConflictAlgorithm.replace,
          );
        }
      }
      await batch.commit(noResult: true);
    });
  }

  Future<List<Playlist>> getCachedPlaylists() async {
    try {
      final db = await database;
      final List<Map<String, dynamic>> results = await db.rawQuery('''
        SELECT 
          p.id as p_id, 
          p.title as p_title, 
          t.id as t_id, 
          t.title as t_title, 
          t.file_path, 
          t.duration_ms 
        FROM playlists p 
        LEFT JOIN tracks t ON p.id = t.playlist_id
      ''');

      if (results.isEmpty) return [];

      final Map<String, Playlist> playlistMap = {};
      for (var row in results) {
        final String playlistId = row['p_id'];
        if (!playlistMap.containsKey(playlistId)) {
          playlistMap[playlistId] = Playlist(
            id: playlistId,
            title: row['p_title'],
            tracks: [],
          );
        }

        if (row['t_id'] != null) {
          playlistMap[playlistId]!.tracks.add(
            AudioTrack(
              id: row['t_id'],
              title: row['t_title'],
              filePath: row['file_path'],
              duration: row['duration_ms'] != null
                  ? Duration(milliseconds: row['duration_ms'])
                  : null,
            ),
          );
        }
      }

      return playlistMap.values.toList();
    } catch (e) {
      debugPrint("DatabaseService: Błąd podczas pobierania cache (JOIN): $e");
      return [];
    }
  }

  Future<void> clearCache() async {
    final db = await database;
    await db.transaction((txn) async {
      await txn.delete('tracks');
      await txn.delete('playlists');
    });
  }

  Future<List<AudioTrack>> searchTracks(String query) async {
    if (query.isEmpty) return [];

    try {
      final db = await database;
      final List<Map<String, dynamic>> results = await db.query(
        'tracks',
        where: 'title LIKE ?',
        whereArgs: ['%$query%'],
      );

      return results.map((row) {
        return AudioTrack(
          id: row['id'] as String,
          title: row['title'] as String,
          filePath: row['file_path'] as String,
          duration: row['duration_ms'] != null
              ? Duration(milliseconds: row['duration_ms'] as int)
              : null,
        );
      }).toList();
    } catch (e) {
      debugPrint("DatabaseService: Błąd podczas wyszukiwania utworów: $e");
      return [];
    }
  }
}

/// Klasa reprezentująca pojedynczy utwór audio w aplikacji.
/// Zaprojektowana pod kątem asynchronicznego ładowania metadanych i pełnej kompatybilności
/// z repozytoriami plików.
class AudioTrack {
  /// Unikalny identyfikator utworu (zazwyczaj ścieżka lokalna do pliku).
  final String id;

  /// Tytuł utworu wyekstrahowany z nazwy pliku lub metadanych ID3.
  final String title;

  /// Pełna ścieżka systemowa do zasobu audio (format MP3).
  final String filePath;

  /// Czas trwania utworu. Pole opcjonalne (nullable), ponieważ metadane
  /// są często ładowane asynchronicznie po wstępnym zaindeksowaniu plików.
  final Duration? duration;

  AudioTrack({
    required this.id,
    required this.title,
    required this.filePath,
    this.duration,
  });

  /// Metoda pomocnicza do tworzenia kopii obiektu z opcjonalną zmianą pól.
  /// Obsługuje architekturę asynchroniczną, pozwalając na aktualizację obiektu
  /// o brakujące metadane (np. czas trwania) bez utraty pozostałych informacji.
  AudioTrack copyWith({
    String? id,
    String? title,
    String? filePath,
    Duration? duration,
  }) {
    return AudioTrack(
      id: id ?? this.id,
      title: title ?? this.title,
      filePath: filePath ?? this.filePath,
      duration: duration ?? this.duration,
    );
  }
}

enum SortMode { original, alphabetical, duration }

class PlaylistDetailsScreen extends StatefulWidget {

  final Playlist playlist;

  @override
  State<PlaylistDetailsScreen> createState() => _PlaylistDetailsScreenState();
}

class _PlaylistDetailsScreenState extends State<PlaylistDetailsScreen> {
  int _selectedFilterIndex = 0;
  SortMode _selectedSortMode = SortMode.alphabetical;
  bool _isAscending = true;

  // Memoizacja listy utworów
  List<AudioTrack>? _memoizedTracks;
  int? _lastFilterIndex;
  SortMode? _lastSortMode;
  bool? _lastAscending;

  final List<String> _filters = ["Wszystkie", "< 5 min", "5 - 15 min", "> 15 min"];

  String _getSortLabel(SortMode mode) {
    switch (mode) {
      case SortMode.alphabetical:
        return "Alfabetycznie";
      case SortMode.duration:
        return "Czas trwania";
      case SortMode.original:
        return "";
    }
  }

  int _naturalCompare(String a, String b) {
    final alphaNum = RegExp(r'(\d+)|(\D+)');
    final aMatches = alphaNum.allMatches(a.toLowerCase()).map((m) => m.group(0)!).toList();
    final bMatches = alphaNum.allMatches(b.toLowerCase()).map((m) => m.group(0)!).toList();

    for (int i = 0; i < aMatches.length && i < bMatches.length; i++) {
      final aPart = aMatches[i];
      final bPart = bMatches[i];

      if (aPart != bPart) {
        final aInt = int.tryParse(aPart);
        final bInt = int.tryParse(bPart);

        if (aInt != null && bInt != null) {
          return aInt.compareTo(bInt);
        }
        return aPart.compareTo(bPart);
      }
    }
    return aMatches.length.compareTo(bMatches.length);
  }

  List<AudioTrack> get _filteredTracks {
    // Sprawdzenie czy parametry się zmieniły, jeśli nie - zwróć cache
    if (_memoizedTracks != null &&
        _lastFilterIndex == _selectedFilterIndex &&
        _lastSortMode == _selectedSortMode &&
        _lastAscending == _isAscending) {
      return _memoizedTracks!;
    }

    List<AudioTrack> tracks = List.from(widget.playlist.tracks);

    // Filtrowanie czasowe
    if (_selectedFilterIndex != 0) {
      tracks = tracks.where((track) {
        final minutes = track.duration?.inMinutes ?? 0;
        switch (_selectedFilterIndex) {
          case 1:
            return minutes < 5;
          case 2:
            return minutes >= 5 && minutes <= 15;
          case 3:
            return minutes > 15;
          default:
            return true;
        }
      }).toList();
    }

    // Sortowanie
    tracks.sort((a, b) {
      int result;
      switch (_selectedSortMode) {
        case SortMode.alphabetical:
          result = _naturalCompare(a.title, b.title);
          break;
        case SortMode.duration:
          final durA = a.duration ?? Duration.zero;
          final durB = b.duration ?? Duration.zero;
          result = durA.compareTo(durB);
          break;
        case SortMode.original:
        default:
          result = 0;
          break;
      }
      return _isAscending ? result : -result;
    });

    // Aktualizacja cache'u
    _memoizedTracks = tracks;
    _lastFilterIndex = _selectedFilterIndex;
    _lastSortMode = _selectedSortMode;
    _lastAscending = _isAscending;

    return tracks;
  }

  String _formatDuration(Duration duration) {
    String twoDigits(int n) => n.toString().padLeft(2, "0");
    String twoDigitMinutes = twoDigits(duration.inMinutes.remainder(60));
    String twoDigitSeconds = twoDigits(duration.inSeconds.remainder(60));
    if (duration.inHours > 0) {
      return "${duration.inHours}:$twoDigitMinutes:$twoDigitSeconds";
    } else {
      return "$twoDigitMinutes:$twoDigitSeconds";
    }
  }

  Future<void> _playTrack(BuildContext context, AudioTrack track, List<AudioTrack> currentList) async {
    if (!context.mounted) return;
    // Haptic Feedback przed nawigacją
    await HapticFeedback.lightImpact();

    Navigator.pushNamed(context, PlayerScreen.routeName, arguments: track);
    try {
      final audioService = ZenAudioService();
      await audioService.loadPlaylist(currentList, currentList.indexOf(track), playlistName: widget.playlist.title);
      await audioService.play();
    } catch (e) {
      debugPrint('Błąd podczas inicjacji odtwarzania: $e');
    }
  }

                          ),
                        ),
                        subtitle: Text(
                          track.duration != null ? _formatDuration(track.duration!) : "--:--",

                        trailing: Icon(

  Future<void> _checkPermissions() async {
    if (kIsWeb) {
      _onPermissionGranted();
      return;
    }

    try {
      PermissionStatus audioStatus = await Permission.audio.status;
      PermissionStatus storageStatus = await Permission.storage.status;
      PermissionStatus notificationStatus = await Permission.notification.status;

      bool audioStorageGranted = audioStatus.isGranted || storageStatus.isGranted;
      bool notificationsGranted = notificationStatus.isGranted;

      if (audioStorageGranted && notificationsGranted) {
        _onPermissionGranted();
      } else {
        _requestPermission();
      }
    } catch (e) {
      debugPrint("Błąd sprawdzania uprawnień: $e");
      _onPermissionGranted();
    }
  }

  Future<void> _requestPermission() async {
    if (!mounted) return;
    setState(() => _isLoading = true);

    if (kIsWeb) {
      _onPermissionGranted();
      return;
    }

    try {
      Map<Permission, PermissionStatus> statuses = await [
        Permission.audio,
        Permission.storage,
        Permission.notification,
      ].request();

      bool audioStorageGranted = statuses[Permission.audio] == PermissionStatus.granted ||
          statuses[Permission.storage] == PermissionStatus.granted;
      bool notificationsGranted = statuses[Permission.notification] == PermissionStatus.granted;

      if (!mounted) return;
      if (audioStorageGranted && notificationsGranted) {
        _onPermissionGranted();
      } else {
        setState(() {
          _isLoading = false;
          _hasPermission = false;
        });
      }
    } catch (e) {
      debugPrint("Błąd prośby o uprawnienia: $e");
      if (!mounted) return;
      setState(() {
        _isLoading = false;
        _hasPermission = false;
      });
    }
  }

  void _onPermissionGranted() {
    if (!mounted) return;
    setState(() {
      _isLoading = false;
      _hasPermission = true;
    });
  }

  String _formatDuration(Duration duration) {
    String twoDigits(int n) => n.toString().padLeft(2, "0");
    String twoDigitMinutes = twoDigits(duration.inMinutes.remainder(60));
    String twoDigitSeconds = twoDigits(duration.inSeconds.remainder(60));
    if (duration.inHours > 0) {
      return "${duration.inHours}:$twoDigitMinutes:$twoDigitSeconds";
    } else {
      return "$twoDigitMinutes:$twoDigitSeconds";
    }
  }

class SeekerSlider extends StatefulWidget {
  final ZenAudioService audioService;

  const SeekerSlider({
    super.key,
    required this.audioService,
  });

  @override
  State<SeekerSlider> createState() => _SeekerSliderState();
}

class _SeekerSliderState extends State<SeekerSlider> {
  bool _isDragging = false;
  double _dragValue = 0.0;

  String _formatDuration(Duration duration) {
    String twoDigits(int n) => n.toString().padLeft(2, "0");
    String twoDigitMinutes = twoDigits(duration.inMinutes.remainder(60));
    String twoDigitSeconds = twoDigits(duration.inSeconds.remainder(60));
    return "${duration.inHours > 0 ? '${duration.inHours}:' : ''}$twoDigitMinutes:$twoDigitSeconds";
  }

  @override
  Widget build(BuildContext context) {
    return StreamBuilder<Duration>(
      stream: widget.audioService.positionStream,
      builder: (context, snapshot) {
        final position = snapshot.data ?? Duration.zero;
        return StreamBuilder<Duration?>(
          stream: widget.audioService.durationStream,
          builder: (context, durationSnapshot) {
            final duration = durationSnapshot.data ?? Duration.zero;
            final maxVal = duration.inMilliseconds.toDouble();

            // Zapewnienie, że wartość suwaka mieści się w zakresie
            double sliderValue = _isDragging
                ? _dragValue.clamp(0.0, maxVal > 0 ? maxVal : 1.0)
                : position.inMilliseconds.toDouble().clamp(0.0, maxVal > 0 ? maxVal : 1.0);

            return Column(
              children: [
                SliderTheme(
                  data: SliderTheme.of(context).copyWith(
                    trackHeight: 3.0,
                    activeTrackColor: Colors.tealAccent,
                    inactiveTrackColor: Colors.white.withOpacity(0.05),
                    thumbColor: Colors.white,
                    overlayColor: Colors.tealAccent.withOpacity(0.1),
                    thumbShape: const RoundSliderThumbShape(enabledThumbRadius: 6.0),
                  ),
                  child: Slider(
                    min: 0.0,
                    max: maxVal > 0 ? maxVal : 1.0,
                    value: sliderValue,
                    onChanged: (value) {
                      setState(() {
                        _isDragging = true;
                        _dragValue = value;
                      });
                    },
                    onChangeEnd: (value) {
                      widget.audioService.seek(Duration(milliseconds: value.toInt()));
                      setState(() {
                        _isDragging = false;
                      });
                    },
                  ),
                ),
                Padding(
                  padding: const EdgeInsets.symmetric(horizontal: 16.0),
                  child: Row(
                    mainAxisAlignment: MainAxisAlignment.spaceBetween,
                    children: [
                      Text(
                        _formatDuration(position),
                        style: GoogleFonts.outfit(
                          color: Colors.white38,
                          fontSize: 12,
                        ),
                      ),
                      Text(
                        _formatDuration(duration),
                        style: GoogleFonts.outfit(
                          color: Colors.white38,
                          fontSize: 12,
                        ),
                      ),
                    ],
                  ),
                ),
              ],
            );
          },
        );
      },
    );
  }
}

const fs = require('fs');
const path = require('path');
const chokidar = require('chokidar');

// Domyślna konfiguracja (fallback)
let CONFIG = {
    watchFolder: './',
    outputFile: 'context_bundle.txt',
    allowedExtensions: ['.js', '.jsx', '.ts', '.tsx', '.css', '.html', '.json', '.py', '.md'],
    ignorePatterns: ['node_modules', '.git', 'dist', 'build', 'package-lock.json']
};

try {
    if (fs.existsSync('watcher.config.json')) {
        const userConfig = JSON.parse(fs.readFileSync('watcher.config.json', 'utf8'));
        CONFIG = { ...CONFIG, ...userConfig };
    }
    
    // ZABEZPIECZENIE: Zawsze dodaj plik wyjściowy do ignorowanych
    if (!CONFIG.ignorePatterns.includes(CONFIG.outputFile)) {
        CONFIG.ignorePatterns.push(CONFIG.outputFile);
    }
    
} catch (e) {
    console.warn('⚠️ Nie udało się wczytać configu, używam domyślnego.');
}

let debounceTimer;

const mergeFiles = () => {
    console.log('🔄 Wykryto zmiany. Generowanie kontekstu...');
    let fullContent = `--- PROJECT CONTEXT GENERATED AT ${new Date().toLocaleString()} ---\n\n`;
    
    fullContent += `// --- Configuration used: watcher.config.json ---\n`;
    fullContent += JSON.stringify(CONFIG, null, 2) + '\n\n';

    const walkSync = (dir, filelist = []) => {
        const files = fs.readdirSync(dir);
        files.forEach((file) => {
            const filePath = path.join(dir, file);
            const stat = fs.statSync(filePath);
            
            if (CONFIG.ignorePatterns.some(pattern => filePath.includes(pattern))) return;

            if (stat.isDirectory()) {
                walkSync(filePath, filelist);
            } else {
                if (CONFIG.allowedExtensions.includes(path.extname(file))) {
                    filelist.push(filePath);
                }
            }
        });
        return filelist;
    };

    try {
        const allFiles = walkSync(CONFIG.watchFolder);
        allFiles.forEach(filePath => {
            const content = fs.readFileSync(filePath, 'utf8');
            const relativePath = path.relative('.', filePath).replace(/\\/g, '/');
            fullContent += `\n// ==========================================\n// --- File: ${relativePath} ---\n// ==========================================\n\n${content}\n`;
        });

        fs.writeFileSync(CONFIG.outputFile, fullContent);
        console.log(`✅ Zaktualizowano: ${CONFIG.outputFile} (${allFiles.length} plików)`);
    } catch (err) {
        console.error('❌ Błąd:', err.message);
    }
};

const watcher = chokidar.watch(CONFIG.watchFolder, {
    ignored: (path) => CONFIG.ignorePatterns.some(s => path.includes(s)),
    persistent: true,
    ignoreInitial: true
});

const handleEvent = () => {
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(() => mergeFiles(), 300);
};

watcher.on('all', handleEvent);

console.log(`👀 Watcher uruchomiony. Obserwuję folder: ${CONFIG.watchFolder}`);
console.log(`📝 Plik wynikowy: ${CONFIG.outputFile}`);
console.log(`💡 Edytuj 'watcher.config.json' aby zmienić ustawienia.`);
mergeFiles();
